from typing import List, Optional, Union
from pydantic import BaseModel, Field

class FXInputItem(BaseModel):
    asset_class: str = Field(description="The category of investment, e.g., Equities, Real Estate")
    is_fx_required: bool = Field(description="True if documented currency differs from reference currency and sufficient data exists")
    
    # Financial Details
    documented_currency: Optional[str] = Field(None, description="The currency found in the text (e.g., USD)")
    reference_currency: Optional[str] = Field(None, description="The target currency (e.g., SGD)")
    
    # Temporal Logic
    year_start: Optional[int] = Field(None, description="The start year for calculation")
    year_end: Optional[int] = Field(None, description="The end year. Matches year_start if single year.")
    is_tenure_average: bool = Field(description="True if this represents a range/tenure requiring an average rate")
    
    # Context for Remarks
    gain_amount: Optional[str] = Field(None, description="The raw gain string, e.g., '+USD 546k'")
    asset_description: Optional[str] = Field(None, description="Brief context of the asset section")

class SOWExtraction(BaseModel):
    sections: List[FXInputItem] = Field(description="List of all identified gain components across all asset classes")
